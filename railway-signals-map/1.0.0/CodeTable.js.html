    /**
 * 代表码位表。每一个站场图有一套对应的码位表，没有站场图的码位表没有意义。码位表对当前站场图的所有（信号机信号、红白光带信号）的全部状态（主要是颜色状态）进行编码。此后，后端将依据这套编码规则定时向前端发送一组编号，前端根据码位表中的编码规则将这些编号转换成可视化语言呈现在页面上。
 * @class
 */
class CodeTable {
  /**
   * 码位表原始数据对象
   */
  #originalData = {};
  /**
   * 当前解析后的编码值
   */
  #currentFrameData = [];
  /**
   * 初始化一个码位表实例
   * @param {*} data
   */
  constructor() {}
  /**
   * 设置码位表数据
   * @param {*} data
   */
  set(data) {
    /**
     * 每次设置前先置空
     */
    this.#originalData = {};
    for (let key in data) {
      const value = data[key];
      /**
       * 过滤掉值为 NULL 的码位
       */
      if (value === "NULL") continue;
      this.#originalData[key] = value;
    }
  }
  /**
   * 按照一组编号从码位表解析出需要变化的颜色值
   * @param {*} numbers - 一组编号
   * @returns numbers
   */
  parse(numbers = []) {
    return numbers
      .map((num) => {
        return this.#originalData[num];
      })
      .filter((d) => d);
  }
}

export default CodeTable;
